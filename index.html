// --------------------------------------------------------- // V19.29 ENGINE PATCH: Import, Master Mock, and Rename // --------------------------------------------------------- // 1. Trigger the hidden file input when "Add" is clicked window.triggerImport = (subject) => { requireAuth(() => { targetImportSubject = subject; // Remember which subject card we clicked document.getElementById('global-file-input').click(); }); }; // 2. Listen for the file upload and inject it into the Master DB document.addEventListener('DOMContentLoaded', () => { document.getElementById('global-file-input').addEventListener('change', function(e) { const file = e.target.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = function(event) { try { const importedData = JSON.parse(event.target.result); const newId = 'quiz_' + Date.now(); // Generate unique ID // Force the subject to match the card where the Add button was clicked importedData.subject = targetImportSubject; db[newId] = importedData; saveData(); renderDashboard(); alert(`✅ Quiz successfully imported into ${targetImportSubject}!`); } catch(err) { alert("❌ Error: Invalid JSON file. Make sure your formatting is perfect."); } e.target.value = ''; // Reset the input so you can upload the same file again if needed }; reader.readAsText(file); }); }); // 3. Generate a "Master Mock" from all questions in a specific subject window.generateSubjectMock = (subject) => { let pool = []; // Gather all questions from this subject across all quizzes Object.keys(db).forEach(quizId => { const quiz = db[quizId]; if (quiz && quiz.subject === subject && Array.isArray(quiz.questions) && !quiz.title.includes("🎯") && !quiz.title.includes("Master Mock") && !quiz.title.includes("SRS")) { pool.push(...quiz.questions); } }); if(pool.length === 0) return alert(`No questions found in ${subject} to create a Master Mock.`); // Deduplicate and Shuffle const uniqueQs = Array.from(new Map(pool.map(q => [q.question, q])).values()); const shuffled = uniqueQs.sort(() => 0.5 - Math.random()); // Cap at 100 questions to prevent browser lag during a massive test const finalQs = shuffled.slice(0, 100); db['quiz_master_mock'] = { title: `🧠 Master Mock - ${subject}`, subject: subject, timeInMinutes: Math.ceil(finalQs.length * 1.5), // Gives 1.5 mins per question questions: finalQs }; openLobby('quiz_master_mock'); }; // 4. Rename Quiz Feature (Updates History linking too) window.renameQuiz = (id) => { requireAuth(() => { const oldTitle = db[id].title; const newTitle = prompt("Enter new name for this quiz:", oldTitle); if (newTitle && newTitle.trim() !== "" && newTitle.trim() !== oldTitle) { const cleanTitle = newTitle.trim(); db[id].title = cleanTitle; // Update history so the "Hist" button doesn't lose track of past attempts if(Array.isArray(historyDb)) { historyDb.forEach(h => { if(h.title === oldTitle) h.title = cleanTitle; }); } saveData(); renderDashboard(); } }); };
